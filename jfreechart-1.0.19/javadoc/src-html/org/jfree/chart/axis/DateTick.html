044package org.jfree.chart.axis;
045
046import java.util.Date;
047
048import org.jfree.chart.util.ParamChecks;
049import org.jfree.ui.TextAnchor;
050import org.jfree.util.ObjectUtilities;
051
052/**
053 * A tick used by the {@link DateAxis} class.
054 */
055public class DateTick extends ValueTick {
056
057    /** The date. */
058    private Date date;
059
060    /**
061     * Creates a new date tick.
062     *
063     * @param date  the date.
064     * @param label  the label.
065     * @param textAnchor  the part of the label that is aligned to the anchor
066     *                    point.
067     * @param rotationAnchor  defines the rotation point relative to the text.
068     * @param angle  the rotation angle (in radians).
069     */
070    public DateTick(Date date, String label,
071                    TextAnchor textAnchor, TextAnchor rotationAnchor,
072                    double angle) {
073        this(TickType.MAJOR, date, label, textAnchor, rotationAnchor, angle);
074    }
075
076    /**
077     * Creates a new date tick.
078     *
079     * @param tickType the tick type (<code>null</code> not permitted).
080     * @param date  the date.
081     * @param label  the label.
082     * @param textAnchor  the part of the label that is aligned to the anchor
083     *                    point.
084     * @param rotationAnchor  defines the rotation point relative to the text.
085     * @param angle  the rotation angle (in radians).
086     *
087     * @since 1.0.12
088     */
089    public DateTick(TickType tickType, Date date, String label,
090                    TextAnchor textAnchor, TextAnchor rotationAnchor,
091                    double angle) {
092        super(tickType, date.getTime(), label, textAnchor, rotationAnchor,
093                angle);
094        ParamChecks.nullNotPermitted(tickType, "tickType");
095        this.date = date;
096    }
097
098    /**
099     * Returns the date.
100     *
101     * @return The date.
102     */
103    public Date getDate() {
104        return this.date;
105    }
106
107    /**
108     * Tests this tick for equality with an arbitrary object.
109     *
110     * @param obj  the object to test (<code>null</code> permitted).
111     *
112     * @return A boolean.
113     */
114    @Override
115    public boolean equals(Object obj) {
116        if (obj == this) {
117            return true;
118        }
119        if (!(obj instanceof DateTick)) {
120            return false;
121        }
122        DateTick that = (DateTick) obj;
123        if (!ObjectUtilities.equal(this.date, that.date)) {
124            return false;
125        }
126        return super.equals(obj);
127    }
128
129    /**
130     * Returns a hash code for this object.
131     *
132     * @return A hash code.
133     */
134    @Override
135    public int hashCode() {
136        return this.date.hashCode();
137    }
138
139}
