042package org.jfree.chart.urls;
043
044import org.jfree.data.xy.XYZDataset;
045
046/**
047 * Interface for a URL generator for plots that uses data from an
048 * {@link XYZDataset}.  Classes that implement this interface are responsible
049 * for correctly escaping any text that is derived from the dataset, as this
050 * may be user-specified and could pose a security risk.
051 */
052public interface XYZURLGenerator extends XYURLGenerator {
053
054    /**
055     * Generates a URL for a particular item within a series. As a guideline,
056     * the URL should be valid within the context of an XHTML 1.0 document.
057     *
058     * @param dataset  the dataset (<code>null</code> not permitted).
059     * @param series  the series index (zero-based).
060     * @param item  the item index (zero-based).
061     *
062     * @return A string containing the generated URL.
063     */
064    public String generateURL(XYZDataset dataset, int series, int item);
065
066}
