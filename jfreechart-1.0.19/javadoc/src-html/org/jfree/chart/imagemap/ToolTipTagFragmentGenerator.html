041package org.jfree.chart.imagemap;
042
043/**
044 * Interface for generating the tooltip fragment of an HTML image map area tag.
045 * The fragment should be <code>XHTML 1.0</code> compliant.
046 */
047public interface ToolTipTagFragmentGenerator {
048
049    /**
050     * Generates a tooltip string to go in an HTML image map.  To allow for
051     * varying standards compliance among browsers, this method is expected
052     * to return an 'alt' attribute IN ADDITION TO whatever it does to create
053     * the tooltip (often a 'title' attribute).
054     * <br><br>
055     * Note that the <code>toolTipText</code> may have been generated from
056     * user-defined data, so care should be taken to filter/escape any
057     * characters that may corrupt the HTML tag.
058     *
059     * @param toolTipText  the tooltip.
060     *
061     * @return The formatted HTML area tag attribute(s).
062     */
063    public String generateToolTipFragment(String toolTipText);
064
065}
