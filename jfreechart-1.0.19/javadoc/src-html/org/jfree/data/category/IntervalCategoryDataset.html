047package org.jfree.data.category;
048
049/**
050 * A category dataset that defines a value range for each series/category
051 * combination.
052 */
053public interface IntervalCategoryDataset extends CategoryDataset {
054
055    /**
056     * Returns the start value for the interval for a given series and category.
057     *
058     * @param series  the series (zero-based index).
059     * @param category  the category (zero-based index).
060     *
061     * @return The start value (possibly <code>null</code>).
062     *
063     * @see #getEndValue(int, int)
064     */
065    public Number getStartValue(int series, int category);
066
067    /**
068     * Returns the start value for the interval for a given series and category.
069     *
070     * @param series  the series key.
071     * @param category  the category key.
072     *
073     * @return The start value (possibly <code>null</code>).
074     *
075     * @see #getEndValue(Comparable, Comparable)
076     */
077    public Number getStartValue(Comparable series, Comparable category);
078
079    /**
080     * Returns the end value for the interval for a given series and category.
081     *
082     * @param series  the series (zero-based index).
083     * @param category  the category (zero-based index).
084     *
085     * @return The end value (possibly <code>null</code>).
086     *
087     * @see #getStartValue(int, int)
088     */
089    public Number getEndValue(int series, int category);
090
091    /**
092     * Returns the end value for the interval for a given series and category.
093     *
094     * @param series  the series key.
095     * @param category  the category key.
096     *
097     * @return The end value (possibly <code>null</code>).
098     *
099     * @see #getStartValue(Comparable, Comparable)
100     */
101    public Number getEndValue(Comparable series, Comparable category);
102
103}
