043package org.jfree.ui;
044
045import java.awt.GradientPaint;
046import java.awt.Shape;
047
048/**
049 * The interface for a class that can transform a <code>GradientPaint</code> to
050 * fit an arbitrary shape.
051 * 
052 * @author David Gilbert
053 */
054public interface GradientPaintTransformer {
055    
056    /**
057     * Transforms a <code>GradientPaint</code> instance to fit some target 
058     * shape.  Classes that implement this method typically return a new
059     * instance of <code>GradientPaint</code>.
060     * 
061     * @param paint  the original paint (not <code>null</code>).
062     * @param target  the reference area (not <code>null</code>).
063     * 
064     * @return A transformed paint.
065     */
066    public GradientPaint transform(GradientPaint paint, Shape target);
067
068}
