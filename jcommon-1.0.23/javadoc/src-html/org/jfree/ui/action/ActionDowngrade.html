044package org.jfree.ui.action;
045
046import javax.swing.Action;
047
048/**
049 * Defines the 2 new constants introduced by Sun in version 1.3 of the J2SDK.
050 *
051 * @author Thomas Morgner
052 */
053public interface ActionDowngrade extends Action {
054
055    /**
056     * The key used for storing a <code>KeyStroke</code> to be used as the
057     * accelerator for the action.
058     */
059    public static final String ACCELERATOR_KEY = "AcceleratorKey";
060
061    /**
062     * The key used for storing an int key code to be used as the mnemonic
063     * for the action.
064     */
065    public static final String MNEMONIC_KEY = "MnemonicKey";
066
067}
