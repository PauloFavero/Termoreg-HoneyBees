044package org.jfree.ui;
045
046import java.awt.Component;
047import java.text.NumberFormat;
048import javax.swing.JTable;
049import javax.swing.SwingConstants;
050import javax.swing.table.DefaultTableCellRenderer;
051
052/**
053 * A table cell renderer that formats numbers with right alignment in each cell.
054 *
055 * @author David Gilbert
056 */
057public class NumberCellRenderer extends DefaultTableCellRenderer {
058
059    /**
060     * Default constructor - builds a renderer that right justifies the 
061     * contents of a table cell.
062     */
063    public NumberCellRenderer() {
064        super();
065        setHorizontalAlignment(SwingConstants.RIGHT);
066    }
067
068    /**
069     * Returns itself as the renderer. Supports the TableCellRenderer interface.
070     *
071     * @param table  the table.
072     * @param value  the data to be rendered.
073     * @param isSelected  a boolean that indicates whether or not the cell is 
074     *                    selected.
075     * @param hasFocus  a boolean that indicates whether or not the cell has 
076     *                  the focus.
077     * @param row  the (zero-based) row index.
078     * @param column  the (zero-based) column index.
079     *
080     * @return the component that can render the contents of the cell.
081     */
082    public Component getTableCellRendererComponent(final JTable table, 
083            final Object value, final boolean isSelected, 
084            final boolean hasFocus, final int row, final int column) {
085
086        setFont(null);
087        final NumberFormat nf = NumberFormat.getNumberInstance();
088        if (value != null) {
089          setText(nf.format(value));
090        }
091        else {
092          setText("");
093        }
094        if (isSelected) {
095            setBackground(table.getSelectionBackground());
096        }
097        else {
098            setBackground(null);
099        }
100        return this;
101    }
102
103}
