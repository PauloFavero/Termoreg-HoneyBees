043package org.jfree.xml;
044
045import java.util.Arrays;
046
047import org.jfree.JCommonInfo;
048import org.jfree.ui.about.Contributor;
049import org.jfree.ui.about.ProjectInfo;
050
051/**
052 * The info implementation for the XML classes of JCommon.
053 *
054 * @author Thomas Morgner
055 */
056public class JCommonXMLInfo extends ProjectInfo {
057    /**
058     * The info singleton.
059     */
060    private static JCommonXMLInfo singleton;
061
062    /**
063     * Returns the single instance of this class.
064     *
065     * @return The single instance of information about the JCommon library.
066     */
067    public static synchronized JCommonXMLInfo getInstance() {
068        if (singleton == null) {
069            singleton = new JCommonXMLInfo();
070        }
071        return singleton;
072    }
073
074
075    /**
076     * Constructs an empty project info object.
077     */
078    private JCommonXMLInfo() {
079        final JCommonInfo info = JCommonInfo.getInstance();
080
081        setName("JCommon-XML");
082        setVersion(info.getVersion());
083        setInfo(info.getInfo());
084        setCopyright(info.getCopyright());
085
086        setLicenceName(info.getLicenceName());
087        setLicenceText(info.getLicenceText());
088
089        setContributors(Arrays.asList(new Contributor[]{
090            new Contributor("David Gilbert", "david.gilbert@object-refinery.com"),
091            new Contributor("Thomas Morgner", "taqua@users.sourceforge.net"),
092            new Contributor("Peter Becker", "-"),
093        }));
094
095        addLibrary(info);
096    }
097}
